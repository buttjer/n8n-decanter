+ // @ts-check

+ /** @typedef {{ name?: string, email?: string, [key: string]: unknown }} ContactRecord */
+ 
+ const items = $input.all();
+ 
+ /** @type {Set<string>} */
+ const seen = new Set();
+ /** @type {{ json: ContactRecord }[]} */
+ const output = [];
+ 
+ for (const item of items) {
+   /** @type {ContactRecord} */
+   const record = { ...item.json };
+ 
+   if (typeof record.name === "string") {
+     record.name = record.name.trim();
+   }
+   if (typeof record.email === "string") {
+     record.email = record.email.trim().toLowerCase();
+   }
+ 
+   const key = JSON.stringify(record, Object.keys(record).sort());
+   if (seen.has(key)) continue;
+   seen.add(key);
+ 
+   output.push({ json: record });
+ }
+ 
+ return output.map((item) => ({ json: item.json }));
